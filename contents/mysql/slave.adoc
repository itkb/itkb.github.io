= How to setup a mysql slave
See [http://www.percona.com/doc/percona-xtrabackup/2.1/howtos/setting_up_replication.html]

= Start

[source, mysql]
START SLAVE;

= Stop

[source, mysql]
STOP SLAVE;

= Status

[source, mysql]
SHOW SLAVE STATUS \G;

= Make slave jump current instruction

[source, mysql]
STOP SLAVE;
SET GLOBAL sql_slave_skip_counter = 1;
START SLAVE;
